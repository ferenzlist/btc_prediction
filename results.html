m = 10 

def S(a, sample = training_sample):
    Sum = 0
    Length = len(sample)
    for l in range(m, Length):
        x = sample[l-m:l]
        y = sample[l]
        Sum += abs(np.dot(x, a) - y)
    return Sum/Length

a = minimize(S, np.zeros(m), bounds=bound, constraints=lc)
